Verkettung & Hintereinanderausführung

Überblick: Wenn verschiedene Operationen kombiniert werden, müssen ihre Komplexitäten korrekt zusammengeführt werden. Diese Seite zeigt, wie man die Gesamtkomplexität von sequenziellen und verschachtelten Operationen berechnet.

Wichtige Regeln:

  • Sequenzielle Operationen (nacheinander): Addiere die Komplexitäten, behalte die dominante = O(n) + O(n²) → O(n²)
  • Verschachtelte Operationen (ineinander): Multipliziere die Komplexitäten = O(n) * O(n) → O(n²)
  • Konstanten ignorieren: O(2n) = O(n), O(n/2) = O(n)
  • Dominante Terme behalten: O(n² + n + 1) = O(n²)
Beispiel 1: Sequenzielle Operationen (Hintereinander)
function processData(arr) { // Operation 1: Linearer Durchlauf for (let i = 0; i < arr.length; i++) { // O(n) print(arr[i]); } // Operation 2: Weiterer linearer Durchlauf for (let i = 0; i < arr.length; i++) { // O(n) arr[i] = arr[i] * 2; } // Operation 3: Konstante Operation return arr[0]; // O(1) }

function findPairs(arr) { // Äußere Schleife for (let i = 0; i < arr.length; i++) { // O(n) // Innere Schleife (verschachtelt) for (let j = 0; j < arr.length; j++) { // O(n) print(arr[i], arr[j]); } } }
